import React, { useState } from 'react';
import styles from '../assets/styles/gallery.module.css';

/**
 * Componente de galería de imágenes que muestra un conjunto de imágenes. 
 * Al hacer clic en una imagen, se muestra un texto superpuesto sobre la imagen seleccionada.
 * 
 * Incluye:
 * - Un conjunto de imágenes que pueden ser activadas para mostrar un texto superpuesto.
 * - Un overlay que cierra la vista de la imagen activa al hacer clic fuera de ella.
 * 
 * @returns {JSX.Element} - Elemento JSX que representa la galería de imágenes.
 */
const Gallery = () => {
  const [activeIndex, setActiveIndex] = useState(null);

  /**
   * Maneja el clic en una imagen, activando o desactivando su visualización.
   * 
   * @param {number} index - El índice de la imagen que se ha hecho clic.
   */
  const handleImageClick = (index) => {
    setActiveIndex(index === activeIndex ? null : index);
  };

  return (
    <div className={styles.galleryContainer}>
      {images.map((image, index) => (
        <div
          className={`${styles.imageContainer} ${activeIndex === index ? styles.active : ''} ${activeIndex !== null && activeIndex !== index ? styles.inactive : ''}`}
          key={index}
          onClick={() => handleImageClick(index)}
        >
          <img src={image.src} alt={`Gallery image ${index + 1}`} className={styles.image} />
          {activeIndex === index && (
            <div className={styles.textOverlay}>
              <p>{image.text}</p>
            </div>
          )}
        </div>
      ))}
      {activeIndex !== null && (
        <div className={styles.overlay} onClick={() => setActiveIndex(null)}></div>
      )}
    </div>
  );
};

export default Gallery;
