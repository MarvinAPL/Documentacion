import { sendPasswordResetEmail } from "firebase/auth"; // Importa la función para enviar el email de reinicio de contraseña
import React from "react"; // Importa React
import { auth } from "../firebase/config"; // Importa la configuración de autenticación de Firebase
import { useNavigate } from "react-router-dom"; // Importa el hook para navegación

/**
 * Componente ForgotPassword - Permite al usuario solicitar un reinicio de contraseña.
 * 
 * Este componente proporciona un formulario donde el usuario puede ingresar su correo electrónico
 * para solicitar el envío de un enlace para restablecer la contraseña. Después de enviar el correo,
 * el usuario es redirigido a la página principal.
 * 
 * @returns {JSX.Element} - Retorna el JSX para mostrar el formulario de reinicio de contraseña.
 */
export function ForgotPassword() {
    const history = useNavigate(); // Hook para la navegación

    /**
     * Maneja el envío del formulario para reiniciar la contraseña.
     * 
     * Envía un correo electrónico de reinicio de contraseña al usuario y maneja los posibles errores.
     * Redirige al usuario a la página principal después de un envío exitoso.
     * 
     * @param {React.FormEvent<HTMLFormElement>} e - El evento del formulario.
     */
    const handleSubmit = async (e) => {
        e.preventDefault(); // Previene el comportamiento por defecto del formulario
        const emailVal = e.target.email.value; // Obtiene el valor del campo de email
        sendPasswordResetEmail(auth, emailVal)
            .then(() => {
                alert("Check your gmail"); // Muestra un mensaje de éxito
                history("/"); // Redirige a la página principal
            })
            .catch((err) => {
                alert(err.code); // Muestra un mensaje de error
            });
    };

    return (
        <div className="App">
            <h1>Forgot Password</h1>
            <form onSubmit={handleSubmit}>
                <input name="email" type="email" placeholder="Enter your email" required /><br /><br />
                <button type="submit">Send</button>
            </form>
        </div>
    );
}
