import React from 'react'; // Importa React
import { useTranslation } from 'react-i18next'; // Hook para la traducción
import { auth } from '../firebase/config'; // Importa la configuración de autenticación de Firebase
import { toast } from 'react-toastify'; // Importa el módulo para notificaciones

/**
 * Componente EmailVerification - Verifica si el correo electrónico del usuario ha sido confirmado.
 * 
 * Este componente permite al usuario verificar su correo electrónico después de hacer clic en el enlace
 * de confirmación enviado por email. Muestra un mensaje de éxito o error basado en el estado de la verificación.
 * 
 * @returns {JSX.Element} - Retorna el JSX para mostrar la verificación del correo electrónico.
 */
export function EmailVerification() {
    const { t } = useTranslation('EmailVerification'); // Hook para la traducción de texto

    /**
     * Maneja la verificación del correo electrónico del usuario.
     * 
     * Recupera al usuario actual, recarga sus datos y verifica si el correo electrónico ha sido confirmado.
     * Muestra notificaciones dependiendo del estado de verificación.
     */
    const handleCheckVerification = async () => {
        const user = auth.currentUser; // Obtiene el usuario actual
        if (user) {
            await user.reload(); // Recarga los datos del usuario
            if (user.emailVerified) { // Verifica si el correo electrónico está confirmado
                toast.success(t('emailVerified'), {
                    position: "top-center",
                });
                window.location.href = "/login"; // Redirige a la página de inicio de sesión
            } else {
                toast.error(t('emailNotVerified'), {
                    position: "top-center",
                });
            }
        }
    };

    return (
        <div>
            <h2>{t('VERIFICA TU CORREO')}</h2>
            <p>{t('Entra a tu Correo Electrónico y visita el enlace, una vez hecho presiona el botón para verificar tu cuenta')}</p>
            <button onClick={handleCheckVerification}>{t('Correo Verificado')}</button>
        </div>
    );
}
