import React from 'react';
import Modal from 'react-modal';
import styles from "../assets/styles/Profiles.module.css";
// import PropTypes from 'prop-types'; // Descomentar si decides usar PropTypes

/**
 * Componente de modal para editar el perfil del usuario.
 * 
 * Permite al usuario actualizar su nombre, correo electrónico, número de teléfono y foto de perfil.
 * 
 * @param {Object} props - Propiedades del componente.
 * @param {boolean} props.isOpen - Determina si el modal está abierto o cerrado.
 * @param {function} props.onRequestClose - Función que se llama cuando se solicita cerrar el modal.
 * @param {Object} props.userDetails - Detalles del usuario (no utilizados directamente en el componente, pero se incluyen para referencia).
 * @param {string} props.newName - Nombre actualizado del usuario.
 * @param {function} props.setNewName - Función para actualizar el nombre del usuario.
 * @param {string} props.newEmail - Correo electrónico actualizado del usuario.
 * @param {function} props.setNewEmail - Función para actualizar el correo electrónico del usuario.
 * @param {string} props.newPhone - Número de teléfono actualizado del usuario.
 * @param {function} props.setNewPhone - Función para actualizar el número de teléfono del usuario.
 * @param {File} props.newPhoto - Foto de perfil actualizada del usuario.
 * @param {function} props.setNewPhoto - Función para actualizar la foto de perfil del usuario.
 * @param {function} props.handleSaveProfile - Función para guardar los cambios realizados en el perfil.
 * 
 * @returns {JSX.Element} - Modal para editar el perfil del usuario.
 */
const EditProfileModal = ({
  isOpen,
  onRequestClose,
  newName,
  setNewName,
  newEmail,
  setNewEmail,
  newPhone,
  setNewPhone,
  newPhoto,
  setNewPhoto,
  handleSaveProfile
}) => {
  return (
    <Modal
      isOpen={isOpen}
      onRequestClose={onRequestClose}
      contentLabel="Edit Profile"
      className={styles.modalContent}
      overlayClassName={styles.modalOverlay}
    >
      <div className={styles.formE}>
        <h2>Editar Perfil</h2>
        
        <div className={styles.contentI}>
          <label htmlFor="nameInput">Nombre de usuario</label>
          <input
            id="nameInput"
            type="text"
            className={styles.formControl}
            value={newName}
            onChange={(e) => setNewName(e.target.value)}
            placeholder="Nuevo nombre"
          />
        </div>
          
        <div className={styles.contentI}>
          <label htmlFor="emailInput">Correo electrónico</label>
          <input
            id="emailInput"
            type="email"
            className={styles.formControl}
            value={newEmail}
            onChange={(e) => setNewEmail(e.target.value)}
            placeholder="Nuevo correo"
          />
        </div>

        <div className={styles.contentI}>
          <label htmlFor="phoneInput">Número de teléfono</label>
          <input
            id="phoneInput"
            type="text"
            className={styles.formControl}
            value={newPhone}
            onChange={(e) => setNewPhone(e.target.value)}
            placeholder="Nuevo teléfono"
          />
        </div>

        <div className={styles.contentI}>
          <label htmlFor="photoInput">Foto de perfil</label>
          <input
            id="photoInput"
            type="file"
            className={`${styles.formControlI} ${styles.fileInput}`}
            onChange={(e) => setNewPhoto(e.target.files[0])}
          />
        </div>
        
        <div className={styles.contentI}>
          <button className={`${styles.btn1}`} onClick={handleSaveProfile}>
            Guardar
          </button>
        </div>
      </div>
    </Modal>
  );
};
