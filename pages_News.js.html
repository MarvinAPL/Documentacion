import { Navbar } from '../components/navbar1';
import { Footer } from '../components/Footer';
import { useEffect, useState } from 'react';
import pedirObjetos from '../components/pedirObjetos';
import ItemList from '../components/ItemList';
import styles from "../assets/styles/News.module.css";
import Publicate from '../components/Publications';
import { useAuth } from '../components/statusUser';
import { useTranslation } from 'react-i18next';

/**
 * Componente para mostrar noticias y permitir a los administradores agregar nuevas.
 *
 * @component
 * @example
 * return (
 *   <News />
 * )
 */
const News = () => {
    const [modalIsOpen, setModalIsOpen] = useState(false); // Estado para controlar la visibilidad del modal
    const [objetos, setObjetos] = useState([]); // Estado para almacenar los objetos de la colección "News"
    const { isAdmin } = useAuth(); // Hook personalizado para obtener el estado de autenticación del usuario
    const { t } = useTranslation('news'); // Hook de traducción para el namespace 'news'

    /**
     * Abre el modal para agregar una nueva noticia.
     */
    const openModal = () => setModalIsOpen(true);

    /**
     * Cierra el modal.
     */
    const closeModal = () => setModalIsOpen(false);

    useEffect(() => {
        const collectionName = "News";
        pedirObjetos(collectionName)
            .then((res) => {
                setObjetos(res);
            })
            .catch((err) => {
                setObjetos([]);
            });
    }, []);

    /**
     * Campos a mostrar en el componente ItemList.
     * 
     * @type {Object}
     * @property {boolean} image - Si se debe mostrar la imagen.
     * @property {boolean} title - Si se debe mostrar el título.
     * @property {boolean} description - Si se debe mostrar la descripción.
     * @property {boolean} city - Si se debe mostrar la ciudad.
     * @property {boolean} state - Si se debe mostrar el estado.
     * @property {boolean} direction - Si se debe mostrar la dirección.
     * @property {boolean} category - Si se debe mostrar la categoría.
     * @property {boolean} price - Si se debe mostrar el precio.
     */
    const displayFields = {
        image: true,
        title: true,
        description: false,
        city: false,
        state: false,
        direction: false,
        category: false,
        price: false
    };

    const collectionName = "News";

    return (
        <div>
            <div className={styles.headerContainer}>
                <div className={styles.navBa}>
                    <Navbar />
                </div>
                <div className={styles.overlay}></div>
                <div className={styles.headerContent}>
                    <h1>{t('novelties')}</h1>
                </div>
            </div>
            
            {isAdmin && (
                <button onClick={openModal} className="btn btn-primary">
                    {t('add_new_project')}
                </button>
            )}
            <ItemList objetos={objetos} displayFields={displayFields} collectionName={collectionName}/>
            <Publicate
                collectionName="News"
                fields={[
                    { name: 'title', label: t('title'), type: 'text', required: true },
                    { name: 'description', label: t('description'), type: 'text', required: true },
                    // Agrega más campos según sea necesario
                ]}
                modalIsOpen={modalIsOpen}
                closeModal={closeModal}
            />
            <Footer />
        </div>
    );
}

export default News;
