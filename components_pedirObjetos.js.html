import { db } from "../firebase/config";
import { collection, getDocs, query, where } from "firebase/firestore";

/**
 * Obtiene una lista de objetos de una colección en Firestore.
 * 
 * @param {string} collectionName - El nombre de la colección en Firestore desde la que se deben obtener los documentos.
 * @param {string|null} [userId=null] - El ID del usuario para filtrar los documentos (opcional). Si se proporciona, se usa para filtrar los documentos por el campo `userId`.
 * @returns {Promise<Object[]>} - Una promesa que resuelve en una lista de objetos. Cada objeto incluye el ID del documento y los datos del documento.
 * @throws {Error} - Lanza un error si ocurre un problema al obtener los documentos.
 * 
 * @example
 * // Obtener todos los documentos de la colección "items"
 * const items = await pedirObjetos("items");
 * 
 * @example
 * // Obtener documentos de la colección "items" filtrados por userId
 * const userItems = await pedirObjetos("items", "user123");
 */
const pedirObjetos = async (collectionName, userId = null) => {
  try {
    const objetos = [];
    const q = userId 
      ? query(collection(db, collectionName), where("userId", "==", userId)) 
      : collection(db, collectionName);
      
    const querySnapshot = await getDocs(q);

    querySnapshot.forEach((doc) => {
      objetos.push({
        id: doc.id,
        ...doc.data()
      });
    });

    return objetos;
  } catch (error) {
    throw new Error("Error al obtener objetos: " + error.message);
  }
}

export default pedirObjetos;
