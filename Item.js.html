import Card from "./Card";

/**
 * Componente que envuelve y muestra un `Card` con la información proporcionada por el objeto.
 * 
 * @param {Object} props - Las propiedades del componente.
 * @param {Object} props.objeto - Objeto que contiene la información para el card.
 * @param {string} props.objeto.id - ID del objeto.
 * @param {string} props.objeto.image - URL de la imagen del objeto.
 * @param {string} props.objeto.title - Título del objeto.
 * @param {string} props.objeto.description - Descripción del objeto.
 * @param {string} props.objeto.city - Ciudad del objeto.
 * @param {string} props.objeto.state - Estado del objeto.
 * @param {string} props.objeto.direction - Dirección del objeto.
 * @param {string} props.objeto.category - Categoría del objeto.
 * @param {number} props.objeto.price - Precio del objeto.
 * @param {string} props.objeto.userName - Nombre del usuario asociado con el objeto.
 * @param {string} props.objeto.userId - ID del usuario asociado con el objeto.
 * @param {string[]} [props.objeto.additionalImages] - Imágenes adicionales asociadas con el objeto.
 * @param {Object} props.displayFields - Campos a mostrar en el card.
 * 
 * @returns {JSX.Element} - Elemento JSX que representa un card con la información del objeto.
 */
const Item = ({ objeto, collectionName, displayFields }) => {
   return (
      <Card
         id={objeto.id}
         collectionName={collectionName}
         image={objeto.image}
         title={objeto.title}
         description={objeto.description}
         city={objeto.city}
         state={objeto.state}
         direction={objeto.direction}
         category={objeto.category}
         price={objeto.price}
         userName={objeto.userName}
         userId={objeto.userId}
         additionalImages={objeto.additionalImages || []}
         displayFields={displayFields}
      />
   );
};

export default Item;
