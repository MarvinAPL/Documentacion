/**
 * Componente de tarjeta de información para la página de inicio.
 * 
 * Este componente muestra una tarjeta que incluye una imagen, un título, un subtítulo y una descripción.
 * 
 * @param {Object} props - Las propiedades del componente.
 * @param {string} props.title - El título que se mostrará en la tarjeta.
 * @param {string} props.subtitle - El subtítulo que se mostrará debajo del título.
 * @param {string} props.description - La descripción que se mostrará debajo del subtítulo.
 * @param {string} props.image - La URL de la imagen que se mostrará en la tarjeta.
 * 
 * @returns {JSX.Element} - El componente de tarjeta con la imagen, título, subtítulo y descripción.
 */
export function CardInfHome({ title, subtitle, description, image }) {
  return (
    <div className={styles.card}>
      <div className={styles.cardImage}>
        <img src={image} alt={title} />
      </div>
      <div className={styles.cardContent}>
        <h3>{title}</h3>
        <div className={styles.cardHeader}>
          <h4>{subtitle}</h4>
          <p>{description}</p>
        </div> 
      </div>
    </div>
  );
}
