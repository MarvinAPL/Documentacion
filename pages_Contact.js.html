import React, { useRef } from 'react'; // Importa React y el hook useRef
import emailjs from 'emailjs-com'; // Importa la librería para enviar correos
import styles from '../assets/styles/Contact.module.css'; // Importa los estilos específicos para la página Contact
import { Navbar } from '../components/navbar1'; // Importa el componente Navbar
import { Footer } from '../components/Footer'; // Importa el componente Footer
import { useTranslation } from 'react-i18next'; // Importa el hook para la traducción de textos

/**
 * Componente Contact - Página de contacto
 * 
 * Este componente representa la página "Contact" que permite a los usuarios enviar un mensaje
 * a través de un formulario. Incluye una barra de navegación, un formulario para enviar correos,
 * y un pie de página con información de contacto adicional.
 * 
 * @returns {JSX.Element} - Retorna el JSX para renderizar la página.
 */
export function Contact() {
    // Hook para la traducción de textos
    const { t } = useTranslation('Contact');
    // Hook useRef para referirse al formulario
    const form = useRef();

    /**
     * Maneja el envío del formulario de contacto.
     * 
     * @param {Object} e - Evento de envío del formulario.
     */
    const sendEmail = (e) => {
        e.preventDefault(); // Previene el comportamiento por defecto del formulario

        // Envía el formulario usando EmailJS
        emailjs.sendForm('service_zepmoa7', 'template_dkuxlwd', form.current, 'nH6qu4CK3isTQlqZn')
            .then((result) => {
                console.log(result.text); // Imprime el resultado en consola
                alert(t('emailSuccess')); // Muestra un mensaje de éxito
            }, (error) => {
                console.log(error.text); // Imprime el error en consola
                alert(t('emailError')); // Muestra un mensaje de error
            });
    };

    return (
        <div>
            {/* Contenedor del encabezado */}
            <div className={styles.headerContainer}>
                {/* Barra de navegación */}
                <div className={styles.navBa}>
                    <Navbar />
                </div>
                {/* Capa de superposición para efectos visuales */}
                <div className={styles.overlay}></div>
                {/* Contenido del encabezado */}
                <div className={styles.headerContent}>
                    <h1>{t('header.title')}</h1> {/* Título principal */}
                </div>
            </div>
            
            {/* Banner del formulario de contacto */}
            <div className={styles.banner}>
                <h3>{t('header.helpTitle')}</h3>
            </div>
            
            {/* Contenedor del formulario */}
            <div className={styles.formContainer}>
                <form ref={form} onSubmit={sendEmail} className={styles.formulario}>
                    {/* Fila de inputs para nombre y apellidos */}
                    <div className={styles.formRow}>
                        <div className={styles.inputGroup}>
                            <label htmlFor="nombre">{t('form.name')}</label>
                            <input type="text" id="nombre" name="nombre" required className={styles.inputText} placeholder={t('form.placeholderName')} />
                        </div>
                        <div className={styles.inputGroup}>
                            <label htmlFor="apellidos">{t('form.surname')}</label>
                            <input type="text" id="apellidos" name="apellidos" required className={styles.inputText} placeholder={t('form.placeholderSurname')} />
                        </div>
                    </div>
                    {/* Fila de inputs para correo y teléfono */}
                    <div className={styles.formRow}>
                        <div className={styles.inputGroup}>
                            <label htmlFor="correo">{t('form.email')}</label>
                            <input type="email" id="correo" name="correo" required className={styles.inputText} placeholder={t('form.placeholderEmail')} />
                        </div>
                        <div className={styles.inputGroup}>
                            <label htmlFor="telefonos">{t('form.phone')}</label>
                            <input type="text" id="telefonos" name="telefonos" required className={styles.inputText} placeholder={t('form.placeholderPhone')} />
                        </div>
                    </div>
                    {/* Fila de textarea para comentario */}
                    <div className={styles.formRow}>
                        <div className={`${styles.inputGroup} ${styles.fullWidth}`}>
                            <label htmlFor="comentario">{t('form.comment')}</label>
                            <textarea id="comentario" name="comentario" rows="5" required className={styles.textarea}></textarea>
                        </div>
                    </div>
                    {/* Botón de envío del formulario */}
                    <button type="submit" className={styles.submitButton}>{t('form.submitButton')}</button>
                </form>
            </div>

            {/* Información adicional de contacto */}
            <div className={styles.infExtra}>
                {/* Información de dirección */}
                <div className={styles.direccion}>
                    <div className={styles.headerTa}>
                        <h3>Direccion</h3>
                    </div>
                    <div className={styles.texto}>
                        <p>CALLE VENUSTIANO CARRANZA #43 INTERIOR 3, COLONIA CENTRO C. P. 76850, AMEALCO DE BONFIL, QUERÉTARO.</p>
                    </div>
                </div>
                {/* Información de teléfonos */}
                <div className={styles.telefonos}>
                    <div className={styles.telc}>
                        <div>
                            <h3>Telefono fijo</h3>
                        </div>
                        <div>                            
                            <p>448-278-05-40</p>
                        </div>
                    </div>
                    <div className={styles.telc}>
                        <div>
                            <h3>Telefono celular</h3>
                        </div>
                        <div>                            
                            <p>+52 1 427 277 7080</p>
                        </div>
                    </div>
                </div>
            </div>

            {/* Pie de página */}
            <Footer />
        </div>
    );
}
